import { compile, instantiate } from "host:wasm";

// Compile WASM bytes to a module
const moduleId = await compile(wasmBytes);

// Instantiate with optional WASI config
const instance = await instantiate(moduleId, {
  preopens: { "/": "/host/path" },
  env: { "MY_VAR": "value" },
  inheritStdout: true
});

// Call exported function
const result = await instance.call("add", 7, 5);
